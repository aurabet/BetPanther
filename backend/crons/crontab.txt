# Bookmaker Africa - Cron Jobs for Daily Image Generation
# Install with: crontab backend/crons/crontab.txt

# Génération quotidienne à minuit
0 0 * * * cd /var/www/bookmaker && /usr/bin/python3 scripts/daily_generation.py >> logs/daily.log 2>&1

# Génération des matchs du matin à 6h
0 6 * * * cd /var/www/bookmaker && /usr/bin/node backend/crons/morning_matches.js >> logs/morning.log 2>&1

# Génération des matchs de l'après-midi à 12h
0 12 * * * cd /var/www/bookmaker && /usr/bin/node backend/crons/afternoon_matches.js >> logs/afternoon.log 2>&1

# Génération des matchs du soir à 18h
0 18 * * * cd /var/www/bookmaker && /usr/bin/node backend/crons/evening_matches.js >> logs/evening.log 2>&1

# Génération réseaux sociaux toutes les 4 heures
0 */4 * * * cd /var/www/bookmaker && /usr/bin/node backend/crons/social_images.js >> logs/social.log 2>&1

# Nettoyage des vieilles images (garder 7 jours)
0 1 * * * cd /var/www/bookmaker && /usr/bin/node backend/crons/cleanup_images.js >> logs/cleanup.log 2>&1
